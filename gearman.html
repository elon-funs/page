#>0.  
  docker中root 用户不用sudo加.
  sudo apt-get install software-properties-common
  sudo add-apt-repository ppa:gearman-developers/ppa
  sudo apt update
  sudo apt-get install gearman-job-server libgearman-dev
  sudo apt-get upgrade

  sudo wget https://github.com/wcgallego/pecl-gearman/archive/master.zip
  unzip master.zip
  sudo phpize 
  cd pecl-gearman-master
  ./configure
  sudo make && make install  //最后一行是编译后的so文件位置
  php -i | grep php.ini //查看$path/mysql.ini
  echo "extension=gearman.so" > $path/gearman.ini
  php -m | grep gearman //查看是模块是否加入到php扩展,没有的话要重启php再看一次
  terminal-1-cli>gearmd //启动服务manager
  terminal-2-cli>php worker.php　//worker
  terminal-3-cli>php client.php　// client
